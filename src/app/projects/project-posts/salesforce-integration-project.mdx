### Apex Controllers

```apex
public class PatientDataController {
    @AuraEnabled(cacheable=true)
    public static List<Patient__c> getPatientRecords(String providerId) {
        return [
            SELECT Id, Name, Treatment_Status__c, Last_Visit__c 
            FROM Patient__c 
            WHERE Provider__c = :providerId 
            ORDER BY Last_Visit__c DESC
        ];
    }
    
    @AuraEnabled
    public static String updateTreatmentStatus(String patientId, String status) {
        try {
            Patient__c patient = new Patient__c(
                Id = patientId,
                Treatment_Status__c = status
            );
            update patient;
            return 'Success';
        } catch (Exception e) {
            return 'Error: ' + e.getMessage();
        }
    }
}
```
